ví dụ margin-top = margin-bottom
        và margin-left = margin-right

        p {
            margin-top: 10px;
            margin-bottom: 10px;
            margin-left: 20px;
            margin-right: 20px;
        }

        =>
        p {
            margin: 10px 20px;
        }
        
        margin: A B;
        A vừa là giá trị margin-top và margin-bottom
        B vừa là giá trị margin-left và margin-right
